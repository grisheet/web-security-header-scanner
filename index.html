Quick Start
# 1. Install the only dependency
pip install httpx

# 2. Scan a single URL
python header_scanner.py --url https://example.com

# 3. Batch scan from file
python header_scanner.py --input urls.txt

# 4. JSON output (for CI / automation)
python header_scanner.py --url https://example.com --format json

# 5. Tune concurrency and timeout
python header_scanner.py --input urls.txt --concurrency 20 --timeout 15

# 6. Fail CI if any site scores below 80
python header_scanner.py --input urls.txt --fail-under 80
